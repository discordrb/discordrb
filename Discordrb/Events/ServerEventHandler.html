def matches?(event)
  # Check for the proper event type
  return false unless event.is_a? ServerEvent

  [
    matches_all(@attributes[:server], event.server) do |a, e|
      a == if a.is_a? String
             e.name
           elsif a.is_a? Integer
             e.id
           else
             e
           end
    end
  ].reduce(true, &:&)
end

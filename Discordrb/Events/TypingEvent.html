def initialize(data, bot)
  @bot = bot

  @user_id = data['user_id'].to_i

  @channel_id = data['channel_id'].to_i
  @channel = bot.channel(@channel_id)

  @user = if channel.pm?
            channel.recipient
          elsif channel.group?
            bot.user(@user_id)
          else
            bot.member(@channel.server.id, @user_id)
          end

  @timestamp = Time.at(data['timestamp'].to_i)
end

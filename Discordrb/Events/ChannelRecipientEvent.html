def initialize(data, bot)
  @bot = bot

  @channel = bot.channel(data['channel_id'].to_i)
  recipient = data['user']
  recipient_user = bot.ensure_user(recipient)
  @recipient = Discordrb::Recipient.new(recipient_user, @channel, bot)
end

def initialize(data, bot)
  @bot = bot

  @user = bot.user(data['user']['id'].to_i)
  @game = data['game'] ? data['game']['name'] : nil
  @server = bot.server(data['guild_id'].to_i)
  @url = data['game'] ? data['game']['url'] : nil
  @type = data['game'] ? data['game']['type'].to_i : nil
end

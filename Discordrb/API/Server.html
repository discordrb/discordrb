def add_emoji(token, server_id, image, name, reason = nil)
  Discordrb::API.request(
    :guilds_sid_emojis,
    server_id,
    :post,
    "#{Discordrb::API.api_base}/guilds/#{server_id}/emojis",
    { image: image, name: name }.to_json,
    Authorization: token,
    content_type: :json,
    'X-Audit-Log-Reason': reason
  )
end

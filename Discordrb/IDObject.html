def self.synthesise(time)
  ms = (time.to_f * 1000).to_i
  (ms - DISCORD_EPOCH) << 22
end

def creation_time
  # Milliseconds
  ms = (@id >> 22) + DISCORD_EPOCH
  Time.at(ms / 1000.0)
end

def initialize(data, bot, server)
  @bot = bot
  @roles = nil

  @name = data['name']
  @server = server
  @id = data['id'].nil? ? nil : data['id'].to_i
  @animated = data['animated']

  process_roles(data['roles']) if server
end

def ==(other)
  return false unless other.is_a? Emoji
  return Discordrb.id_compare(@id, other) if @id

  name == other.name
end

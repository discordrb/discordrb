def matches?(event)
  # Check for the proper event type
  return false unless event.is_a? RawEvent

  [
    matches_all(@attributes[:type] || @attributes[:t], event.type) do |a, e|
      if a.is_a? Regexp
        a.match?(e)
      else
        e.to_s.casecmp(a.to_s).zero?
      end
    end
  ].reduce(true, &:&)
end

def matches?(event)
  # Check for the proper event type
  return false unless event.is_a? ServerMemberEvent

  [
    matches_all(@attributes[:username], event.user.name) do |a, e|
      a == if a.is_a? String
             e.to_s
           else
             e
           end
    end
  ].reduce(true, &:&)
end

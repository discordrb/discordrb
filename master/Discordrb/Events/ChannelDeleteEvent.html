def initialize(data, bot)
  @bot = bot

  @type = data['type']
  @topic = data['topic']
  @position = data['position']
  @name = data['name']
  @is_private = data['is_private']
  @id = data['id'].to_i
  @server = bot.server(data['guild_id'].to_i) if data['guild_id']
  @owner_id = bot.user(data['owner_id']) if @type == 3
end

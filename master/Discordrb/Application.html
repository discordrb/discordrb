def initialize(data, bot)
  @bot = bot

  @name = data['name']
  @id = data['id'].to_i
  @description = data['description']
  @icon_id = data['icon']
  @rpc_origins = data['rpc_origins']
  @flags = data['flags']
  @owner = @bot.ensure_user(data['owner'])
end
